Q9. What will be the output of the program ?
#include<stdio.h>

int main()
{
    char *names[] = { "Suresh", "Siva", "Sona", "Baiju", "Ritu"};
    int i;
    char *t;
    t = names[3];
    names[3] = names[4];
    names[4] = t;
    for(i=0; i<=4; i++)
        printf("%s,", names[i]);
    return 0;
}
a. Suresh, Siva, Sona, Baiju, Ritu
b. Suresh, Siva, Sona, Ritu, Baiju
c. Suresh, Siva, Baiju, Sona, Ritu
